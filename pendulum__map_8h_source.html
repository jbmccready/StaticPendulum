220 map_type pendulum_map<integrator_type>::create_map_container() const
221 {
222  const int xdim = std::round((m_xend-m_xstart)/m_res)+1;
223  const int ydim = std::round((m_yend-m_ystart)/m_res)+1;
224  int xdim_factor = std::round(m_xstart/m_res); // create int multipliers to fill array to avoid floating math rounding error
225  int ydim_factor = std::round(m_ystart/m_res);
226 
227  map_type integration_map(xdim, std::vector<point_type>(ydim));
228  for (int i = 0; i < xdim; i++) {
229  for (int j = 0; j < ydim; j++) {
230  integration_map[i][j].start_state[0] = double(xdim_factor)*m_res;
231  integration_map[i][j].start_state[1] = double(ydim_factor)*m_res;
232  integration_map[i][j].start_state[2] = 0.0;
233  integration_map[i][j].start_state[3] = 0.0;
234  ydim_factor++;
235  }
236 
237  ydim_factor = std::round(m_ystart/m_res); // reset y value for next column
238  xdim_factor++;
239  }
240 
241  return integration_map;

254 template <typename integrator_type>
255 inline void pendulum_map<integrator_type>::fixed_integrate_point(const integrator_type &the_integrator, const pendulum_system &the_system, point_type &the_point) const
256 {
257  double t = m_tstart;
258  double h = m_dt;
259  the_point.converge_time = m_tend-m_dt;
260  unsigned int trial_count = 0;
261 
262  // check pendulum length boundary
263  if (std::sqrt(std::pow(the_point.start_state[0], 2.0) + std::pow(the_point.start_state[1], 2.0)) < (the_system.L - 1e-10)) {
264  const double absX = std::abs(the_point.start_state[0]);
265  const double absY = std::abs(the_point.start_state[1]);
266  // avoid undefined point at (0,0)
267  if(absX > 1e-10 || absY > 1e-10) {
268  // integration good to go, create local state for integration to keep start state
269  state_type current_state = the_point.start_state;
270  while (t < m_tend && trial_count < 1000000 ) {
271  the_point.step_count += the_integrator.do_step(the_system, current_state, t, h);
272  trial_count++;
273  }
274 
275  for (unsigned int i = 0; i < the_system.attractor_list.size(); i++) {
276  if ((the_system.attractor_list[i].x-m_pos_tol<current_state[0]) && (current_state[0]<the_system.attractor_list[i].x+m_pos_tol) && (the_system.attractor_list[i].y-m_pos_tol<current_state[1]) && (current_state[1]<the_system.attractor_list[i].y+m_pos_tol)) {
277  the_point.converge_position = i;
278  return;
279  }
280  }
281 
282  if ((0.0 - m_mid_tol < current_state[0]) && (current_state[0] < 0.0 + m_mid_tol) && (0.0 - m_mid_tol < current_state[1]) && (current_state[1] < 0.0 + m_mid_tol)) {
283  the_point.converge_position = 254;
284  }
285  }
286  }
287  // ELSE: Position at (0,0) or outside of bounds or not found above attractor, undefined behavior for our pendulum system, leave converge position unset
288 }

315 template <typename integrator_type>
316 inline void pendulum_map<integrator_type>::integrate_point(const integrator_type &the_integrator, const pendulum_system &the_system, point_type &the_point) const
317 {
318  double t = m_tstart;
319  double h = m_dt;
320  unsigned int trial_count = 0;
321 
322  // check pendulum length boundary
323  if (std::sqrt(std::pow(the_point.start_state[0], 2.0) + std::pow(the_point.start_state[1], 2.0)) < (the_system.L - 1e-10)) {
324  const double absX = std::abs(the_point.start_state[0]);
325  const double absY = std::abs(the_point.start_state[1]);
326  // avoid undefined point at (0,0)
327  if(absX > 1e-10 || absY > 1e-10) {
328  // integration good to go, create local state for integration to keep start state and hopefully optimize memory rather than calling a member variable every time
329  state_type current_state = the_point.start_state;
330  bool converged = false;
331  std::vector<double> magnet_time;
332  bool near_magnet = false;
333  unsigned int current_magnet = 0;
334  while (!converged && t < 1000 && trial_count < 1000000 ) {
335  the_point.step_count += the_integrator.do_step(the_system, current_state, t, h);
336  trial_count++;
337  near_magnet = false;
338 
339  // check if pendulum head near an attractor and if it's been near for long enough time to consider converged
340  for (unsigned int i = 0; i < the_system.attractor_list.size(); i++) {
341  if ((the_system.attractor_list[i].x-m_pos_tol<current_state[0]) && (current_state[0]<the_system.attractor_list[i].x+m_pos_tol) && (the_system.attractor_list[i].y-m_pos_tol<current_state[1]) && (current_state[1]<the_system.attractor_list[i].y+m_pos_tol)) {
342  near_magnet = true;
343  if (current_magnet == i) {
344  magnet_time.push_back(t);
345  if (magnet_time.back() - magnet_time.front() >= m_time_tol) {
346  magnet_time.clear();
347  the_point.converge_time = t;
348  the_point.converge_position = i;
349  converged = true;
350  }
351  } else {
352  magnet_time.clear();
353  current_magnet = i;
354  magnet_time.push_back(t);
355  }
356  break;
357  }
358  }
359  if (!near_magnet) {
360  // check if pendulum head near middle
361  if ((0.0 - m_mid_tol < current_state[0]) && (current_state[0] < 0.0 + m_mid_tol) && (0.0 - m_mid_tol < current_state[1]) && (current_state[1] < 0.0 + m_mid_tol)) {
362  near_magnet = true;
363  if (current_magnet == 254) {
364  magnet_time.push_back(t);
365  if (magnet_time.back() - magnet_time.front() >= m_time_tol) {
366  magnet_time.clear();
367  the_point.converge_time = t;
368  the_point.converge_position = 254;
369  converged = true;
370  }
371  } else {
372  magnet_time.clear();
373  current_magnet = 254;
374  magnet_time.push_back(t);
375  }
376  } else {
377  // not near middle or any attractor
378  magnet_time.clear();
379  }
380  }
381  }
382  }
383  }
384  // ELSE: Position at (0,0) or outside of bounds, undefined behavior for our pendulum system, leave converge position unset
385 }
