74 inline void pendulum_system::operator() (const state_type &x, state_type &dxdt, const double /* t */) const
75 {
76  const double x_squared = x[0]*x[0];
77  const double y_squared = x[1]*x[1];
78  const double L_squared = L*L;
79  const double norm_squared = x_squared + y_squared;
80 
81  const double g_value = -m*g/L * sqrt(1.0 - norm_squared/L_squared);
82 
83  double f_m_x = 0.0;
84  double f_m_y = 0.0;
85 
86  const double a_value = (d+L-sqrt(L_squared-norm_squared));
87  const double a_value_squared = a_value*a_value;
88 
89  double ax_value;
90  double ay_value;
91  double ax_value_squared;
92  double ay_value_squared;
93  double a_denom;
94  for (auto attractor : attractor_list) {
95  ax_value = x[0]-attractor.x;
96  ay_value = x[1]-attractor.y;
97  ax_value_squared = ax_value*ax_value;
98  ay_value_squared = ay_value*ay_value;
99  a_denom = -attractor.k/pow(ax_value_squared+ay_value_squared+a_value_squared,1.5);
100 
101  f_m_x += ax_value*a_denom;
102  f_m_y += ay_value*a_denom;
103  }
104 
105  dxdt[0] = x[2];
106  dxdt[1] = x[3];
107  dxdt[2] = (x[0]*g_value - b*x[2] + f_m_x) / m;
